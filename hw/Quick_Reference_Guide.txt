// ESP8266 Relay Control - Basic Example
const int RELAY_PIN = D1;  // GPIO5

void setup() {
  pinMode(RELAY_PIN, OUTPUT);
  digitalWrite(RELAY_PIN, LOW);  // Relay OFF
  Serial.begin(115200);
  Serial.println("ESP8266 Relay Control Ready");
}

void loop() {
  // Turn relay ON
  digitalWrite(RELAY_PIN, HIGH);
  Serial.println("Relay ON");
  delay(2000);
  
  // Turn relay OFF
  digitalWrite(RELAY_PIN, LOW);
  Serial.println("Relay OFF");
  delay(2000);
}

}

void loop() {
  server.handleClient();
}
